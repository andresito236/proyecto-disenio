function ultimasAcciones() {
    $db = Conexion::conectar();
    $sentencia = $db->query("SELECT 'ingreso, IngresoID' AS  Tipo, Fecha, Monto, Descripcion FROM ingreso
      UNION ALL
      SELECT 'egreso, EgresoID' AS Tipo, Fecha, Monto, Descripcion FROM egreso
      ORDER BY Fecha DESC
      LIMIT 3;
    ");
    return $sentencia->fetchAll();
}
